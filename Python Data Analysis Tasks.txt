#Slide 30

import numpy as np

data = [2, 3, 4, 6, 7, 7, 8, 19]

# Method 1: Mean plus or minus 2 standard deviations
mean = np.mean(data)
stddev = np.std(data)

lower_bound_method1 = mean - 2 * stddev
upper_bound_method1 = mean + 2 * stddev

outliers_method1 = [x for x in data if x < lower_bound_method1 or x > upper_bound_method1]

# Method 2: Q3 + 1.5 * IQR and Q1 - 1.5 * IQR
q1 = np.percentile(data, 25)
q3 = np.percentile(data, 75)
iqr = q3 - q1

lower_bound_method2 = q1 - 1.5 * iqr
upper_bound_method2 = q3 + 1.5 * iqr

outliers_method2 = [x for x in data if x < lower_bound_method2 or x > upper_bound_method2]

print("Data:", data)
print("Method 1 Outliers:", outliers_method1)
print("Method 2 Outliers:", outliers_method2)

#Slide 8

import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import binom

n = 10  # Number of trials
p = 0.3  # Probability of success each time

# Define the binomial distribution
binomial_dist = binom(n, p)

# Calculate probabilities for different values of X
x_values = np.arange(0, n+1)  # Values of X: 0, 1, 2, ..., n
probabilities = binomial_dist.pmf(x_values)

# Create a bar plot to visualize the probability distribution
plt.bar(x_values, probabilities)
plt.xlabel('X (Number of Stops  out of 10 Lights)')
plt.ylabel('Probability')
plt.title('Binomial Distribution: Traffic Lights, p = 0.3')
plt.show()

np.random.seed(1)

number_of_samples = 100000
sample_size = 50

sample_means = []
sample_sd_p = []
sample_sd_s = []

for count in range(number_of_samples):
    sample = np.random.choice(population_data, size = sample_size)
    sample_means.append(np.mean(sample))
    sample_sd_p.append(np.std(sample))
    sample_sd_s.append(np.std(sample, ddof = 1)) #setting ddof = 1 means the n-1 formula will be used.
    
print("The sample means average out as", np.mean(sample_means), 'compared to the population:', population_mean)
print("The samples standard deviations (n formula) average out as", 
      np.mean(sample_sd_p),
      'and for the n-1 formula:',
      np.mean(sample_sd_s), 'compared to the population:', population_sd)

Examples of Linear Regression


import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression

# Generate random data
np.random.seed(0)
x = np.random.rand(50) * 10
y = 2 * x + 1 + np.random.randn(50) * 2

# Reshape the data for scikit-learn
x = x.reshape(-1, 1)
y = y.reshape(-1, 1)

# Create a linear regression model
model = LinearRegression()
model.fit(x, y)

# Generate predicted values
x_pred = np.linspace(0, 10, 100).reshape(-1, 1)
y_pred = model.predict(x_pred)

# Calculate standard deviation of residuals
residuals = y - model.predict(x)
std_dev = np.std(residuals)

# Create a scatter plot
plt.figure(figsize=(10, 6))
plt.scatter(x, y, label="Data points", color="blue")
plt.plot(x_pred, y_pred, color="red", label="Linear Regression")
plt.fill_between(
    x_pred.flatten(),
    y_pred.flatten() - 2 * std_dev,
    y_pred.flatten() + 2 * std_dev,
    color="gray",
    alpha=0.3,
    label="Error Window",
)
plt.xlabel("X")
plt.ylabel("Y")
plt.title("Scatter Plot with Linear Regression and Error Window")

# Annotate the plot with the equation of the line
equation = f"y = {model.coef_[0][0]:.2f}x + {model.intercept_[0]:.2f}"
plt.text(2, 18, equation, fontsize=12, color="green")

plt.legend()
plt.grid(True)
plt.show()

import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
from scipy.stats import pearsonr

# Generate random data
np.random.seed(0)
x = np.random.rand(50) * 10
y = 2 * x + 1 + np.random.randn(50) * 2

# Calculate Pearson correlation coefficient and p-value
corr_coeff, p_value = pearsonr(x, y)

# Reshape the data for scikit-learn
x = x.reshape(-1, 1)
y = y.reshape(-1, 1)

# Create a linear regression model
model = LinearRegression()
model.fit(x, y)

# Generate predicted values
x_pred = np.linspace(0, 15, 100).reshape(-1, 1)  # Extend the range
y_pred = model.predict(x_pred)

# Add predicted points for larger x coordinates
new_x = np.array([[10], [14.5],[-5]])  # Example new x coordinates
new_y_pred = model.predict(new_x)


# Create a scatter plot
plt.figure(figsize=(10, 6))
plt.scatter(x, y, label="Original Data points", color="blue")
plt.plot(new_x, new_y_pred, color="purple")

# Add vertical lines at the smallest and largest x coordinates
min_x = np.min(x)
max_x = np.max(x)
plt.axvline(x=min_x, color="green", linestyle="--", label="Smallest and Largest X values in the original data")
plt.axvline(x=max_x, color="green", linestyle="--")


plt.scatter(new_x, new_y_pred, color="magenta", label="New values predicted - Extrapolation")

plt.xlabel("X")
plt.ylabel("Y")

plt.legend()
plt.grid(True)
plt.show()
